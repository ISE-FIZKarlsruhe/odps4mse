<h2 class="odp-section-title" id="propose">Propose a Pattern</h2>

Have a concrete modeling challenge? Describe it using our quick form and we'll create a structured GitHub issue for you.

<div class="odp-quick-form">
  <h3>Quick Pattern Proposal</h3>
  <div class="odp-form-row">
    <div>
      <label for="pf-name">Pattern name</label>
      <input type="text" id="pf-name" placeholder="e.g., Measurement Setup Pattern">
    </div>
    <div>
      <label for="pf-req">Requirement area</label>
      <select id="pf-req">
        <option value="Req 1: Process Structure">Process Structure</option>
        <option value="Req 2: Data & Resources">Data & Resources</option>
        <option value="Req 3: Project & Roles">Project & Roles</option>
        <option value="New requirement">New requirement</option>
      </select>
    </div>
  </div>
  <label for="pf-motivation">What problem does this solve?</label>
  <textarea id="pf-motivation" placeholder="Describe the modeling challenge..."></textarea>
  <div class="odp-form-row">
    <div>
      <label for="pf-name-expert">Your name (optional)</label>
      <input type="text" id="pf-name-expert" placeholder="e.g., Dr. Jane Smith">
    </div>
    <div>
      <label for="pf-affiliation">Affiliation (optional)</label>
      <input type="text" id="pf-affiliation" placeholder="e.g., KIT, TU Dresden...">
    </div>
  </div>
  <button class="odp-submit-btn" onclick="submitProposal()">
    <svg width="18" height="18" viewBox="0 0 24 24" fill="currentColor"><path d="M2.01 21L23 12 2.01 3 2 10l15 2-15 2z"/></svg>
    Open as GitHub Issue
  </button>
</div>

<script>
function submitProposal() {
  const name = document.getElementById('pf-name').value.trim();
  const req = document.getElementById('pf-req').value;
  const motivation = document.getElementById('pf-motivation').value.trim();
  const expert = document.getElementById('pf-name-expert').value.trim();
  const affiliation = document.getElementById('pf-affiliation').value.trim();

  if (!name) { alert('Please enter a pattern name.'); return; }
  if (!motivation) { alert('Please describe the problem this pattern solves.'); return; }

  const title = encodeURIComponent('[Pattern Proposal] ' + name);
  const body = encodeURIComponent(
    '## Pattern: ' + name + '\n\n' +
    '**Requirement area:** ' + req + '\n\n' +
    '### Motivation\n\n' + motivation + '\n\n' +
    (expert ? '**Proposed by:** ' + expert + '\n' : '') +
    (affiliation ? '**Affiliation:** ' + affiliation + '\n' : '') +
    '\n---\n_Submitted via the [Community Hub](https://ise-fizkarlsruhe.github.io/odps4mse/community/)_'
  );

  window.open(
    'https://github.com/ISE-FIZKarlsruhe/odps4mse/issues/new?labels=proposal,pattern&title=' + title + '&body=' + body,
    '_blank'
  );
}
</script>
